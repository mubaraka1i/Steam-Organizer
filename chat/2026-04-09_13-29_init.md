(function () {
  const LANGS = ["en", "es", "zh"];
  let currentIndex = 0; // always start on English per requirement

  // Utility: apply translation map to DOM elements with `data-i18n`.
  function applyTranslations(lang) {
    const map = I18N_KEYS[lang] || I18N_KEYS.en;
    document.querySelectorAll("[data-i18n]").forEach((el) =&gt; {
      const key = el.getAttribute("data-i18n");
      if (!key) return;
      const text = map[key];
      if (typeof text === "undefined") return;
      // Keep markup where appropriate (brand with emoji uses innerHTML elsewhere).
      // Use innerText for safety where possible.
      if (el.tagName === "INPUT" || el.tagName === "TEXTAREA") {
        // If placeholder needs translating in future.
        if (el.placeholder !== undefined) el.placeholder = text;
      } else {
        el.textContent = text;
      }
    });

    // Update document language for accessibility
    document.documentElement.lang = lang;
  }

  // Copy classes from theme-toggle so the language button uses the same animations/styles.
  function inheritButtonClasses() {
    const themeBtn = document.getElementById("theme-toggle");
    const langBtn = document.getElementById("lang-toggle");
    if (!themeBtn || !langBtn) return;
    // copy class list to match any animation/styling on the theme toggle
    langBtn.className = themeBtn.className;
  }

  // Cycle languages on click: EN -&gt; ES -&gt; ZH -&gt; EN ...
  function initLangToggle() {
    const langBtn = document.getElementById("lang-toggle");
    if (!langBtn) return;

    langBtn.addEventListener("click", () =&gt; {
      // small visual feedback: copy/pulse class from theme button if present
      const themeBtn = document.getElementById("theme-toggle");
      const pulseClass = themeBtn ? themeBtn.dataset.pulseClass : null;
      if (pulseClass) {
        langBtn.classList.add(pulseClass);
        setTimeout(() =&gt; langBtn.classList.remove(pulseClass), 250);
      }

      currentIndex = (currentIndex + 1) % LANGS.length;
      const nextLang = LANGS[currentIndex];
      applyTranslations(nextLang);
      // update visible label on the button
      langBtn.textContent = I18N_KEYS[nextLang]["lang.button"] || nextLang.toUpperCase();
    });
  }

  // Initialize on DOM ready
  document.addEventListener("DOMContentLoaded", () =&gt; {
    inheritButtonClasses();
    // Force start on English as requested
    currentIndex = 0;
    applyTranslations(LANGS[currentIndex]);

    // ensure the language button displays the right visible label
    const langBtn = document.getElementById("lang-toggle");
    if (langBtn) langBtn.textContent = I18N_KEYS.en["lang.button"];

    initLangToggle();
  });
})();
